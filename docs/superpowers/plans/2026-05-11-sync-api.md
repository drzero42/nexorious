package platformresolution

// RawPlatformToSlug maps a raw platform string from a sync adapter to a canonical platforms.name slug.
// Returns ("", false) for unknown raw platforms.
func RawPlatformToSlug(raw string) (string, bool) {
	switch raw {
	case "pc-windows":
		return "pc-windows", true
	case "playstation-5":
		return "ps5", true
	case "playstation-4":
		return "ps4", true
	default:
		return "", false
	}
}

// StorefrontToCollectionSlug maps a sync-source storefront identifier to the storefronts.name slug.
// Returns ("", false) for storefronts with no collection mapping (epic, gog, etc.).
func StorefrontToCollectionSlug(storefront string) (string, bool) {
	switch storefront {
	case "steam":
		return "steam", true
	case "psn":
		return "playstation-store", true
	default:
		return "", false
	}
}

func (h *SyncHandler) HandleTriggerSync(c *echo.Context) error {
	userID := auth.UserIDFromContext(c)
	if userID == "" {
		return echo.NewHTTPError(http.StatusUnauthorized, "unauthorized")
	}
	sf := c.Param("storefront")
	if !validTriggerStorefronts[sf] {
		return echo.NewHTTPError(http.StatusBadRequest, "invalid storefront")
	}
	ctx := context.Background()

	// Check for active job.
	var existingID string
	err := h.db.NewRaw(
		`SELECT id FROM jobs WHERE user_id = ? AND job_type = 'sync' AND source = ? AND status IN ('pending', 'processing') LIMIT 1`,
		userID, sf,
	).Scan(ctx, &existingID)
	if err == nil {
		return echo.NewHTTPError(http.StatusConflict, "sync already in progress")
	}
	if !errors.Is(err, sql.ErrNoRows) {
		return echo.NewHTTPError(http.StatusInternalServerError, "failed to check existing job")
	}

	// Insert job.
	now := time.Now().UTC()
	jobID := uuid.NewString()
	_, err = h.db.NewRaw(
		`INSERT INTO jobs (id, user_id, job_type, source, status, priority, created_at) VALUES (?, ?, 'sync', ?, 'pending', 'high', ?)`,
		jobID, userID, sf, now,
	).Exec(ctx)
	if err != nil {
		return echo.NewHTTPError(http.StatusInternalServerError, "failed to create job")
	}

	// Submit worker task.
	if h.pool != nil {
		_ = h.pool.Submit(ctx, "dispatch_sync", map[string]string{
			"job_id": jobID, "user_id": userID, "storefront": sf,
		}, 10)
	}

	return c.JSON(http.StatusOK, manualSyncTriggerResponse{
		Message:    "Sync job created for " + sf,
		JobID:      jobID,
		Storefront: sf,
		Status:     "queued",
	})

	}

	var req struct {
		SteamID   string `json:"steam_id"`
		WebAPIKey string `json:"web_api_key"`
	}
	if err := c.Bind(&req); err != nil {
		return echo.NewHTTPError(http.StatusBadRequest, "invalid body")
	}

	errStr := func(s string) *string { return &s }

	if !steamIDRegex.MatchString(req.SteamID) {
		return c.JSON(http.StatusOK, steamVerifyResponse{Valid: false, Error: errStr("invalid_steam_id")})
	}
	if !steamKeyRegex.MatchString(req.WebAPIKey) {
		return c.JSON(http.StatusOK, steamVerifyResponse{Valid: false, Error: errStr("invalid_api_key")})
	}

	summary, err := h.steamClient.GetPlayerSummaries(c.Request().Context(), req.WebAPIKey, req.SteamID)

		if errors.Is(err, ErrSteamRateLimited) {
			return c.JSON(http.StatusOK, steamVerifyResponse{Valid: false, Error: errStr("rate_limited")})
		}
		return c.JSON(http.StatusOK, steamVerifyResponse{Valid: false, Error: errStr("network_error")})
	}
	if summary == nil {
		return c.JSON(http.StatusOK, steamVerifyResponse{Valid: false, Error: errStr("invalid_steam_id")})
	}
	if summary.CommunityVisibilityState != 3 {
		return c.JSON(http.StatusOK, steamVerifyResponse{Valid: false, Error: errStr("private_profile")})
	}

	creds := map[string]string{
		"web_api_key":  req.WebAPIKey,
		"steam_id":     req.SteamID,
		"display_name": summary.PersonaName,
	}

	credsStr := string(credsJSON)
	now := time.Now().UTC()

The `steam.Client` type satisfies `api.SteamClient` because it has `GetPlayerSummaries(ctx, apiKey, steamID string) (*steam.SteamPlayerSummary, error)` — but `api.SteamClient` expects `*api.SteamPlayerSummary`. These are different types! We need to either:
- Return `*api.SteamPlayerSummary` from the steam client (requires importing `api` package — creates an import cycle if `api` imports `steam`).
- OR: the `steam.Client` satisfies a different interface that has the same shape, wired in router.go via an adapter.

**The correct approach**: the `steam.Client.GetPlayerSummaries` method must return `*api.SteamPlayerSummary` to satisfy `api.SteamClient`. But `api` imports `steam` in `router.go`, and if `steam` also imports `api`, there's a cycle.

**Solution**: define the summary types in a shared package, or have `steam.Client` return its own type and wrap it in `router.go` with an adapter.

**Simpler solution**: define `SteamPlayerSummary` and `PSNAccountInfo` in a separate package (`internal/syncmodels` or similar), import it in both `api` and `steam`. But that's over-engineering.

**Simplest solution**: define the interface in `internal/api/sync.go` with the concrete types. Have `internal/services/steam/client.go` import `internal/api` — but `internal/api/router.go` already imports `internal/services/steam`. That creates a cycle.

**Cleanest solution for this codebase**: define the interfaces and summary types in their own small package `internal/syncclient` imported by both `api` and `steam`. Then `api.SteamClient` becomes `syncclient.SteamClient`, etc.

**However**, the spec explicitly says:
> "Production implementations live in `internal/services/steam/` and `internal/services/psn/`... Test stubs are inline structs in `sync_test.go`."

// dispatchPayload is the PendingTask.Payload shape for "dispatch_sync".
type dispatchPayload struct {
	JobID      string `json:"job_id"`
	UserID     string `json:"user_id"`
	Storefront string `json:"storefront"`
}

// NewDispatchSyncHandler returns a TaskHandler that fetches a user's game library,
// upserts external_games rows, marks removed games, and dispatches process_sync_item tasks.
func NewDispatchSyncHandler(db *bun.DB, steam SteamLibraryAdapter, psnLib PSNLibraryAdapter) func(ctx context.Context, task *models.PendingTask) error {
	return func(ctx context.Context, task *models.PendingTask) error {
		var p dispatchPayload
		if err := json.Unmarshal(task.Payload, &p); err != nil {
			slog.Error("dispatch_sync: unmarshal payload", "err", err)
			return nil
		}

		// ── 1. Mark job processing ────────────────────────────────────────
		now := time.Now().UTC()
		_, _ = db.NewRaw(`UPDATE jobs SET status = 'processing', started_at = ? WHERE id = ?`, now, p.JobID).Exec(ctx)

		// ── 2. Read credentials ───────────────────────────────────────────
		var cfg models.UserSyncConfig
		if err := db.NewSelect().Model(&cfg).Where("user_id = ? AND storefront = ?", p.UserID, p.Storefront).Scan(ctx); err != nil {
			failJob(ctx, db, p.JobID, "no sync config found")
			return nil
		}
		if cfg.StorefrontCredentials == nil {
			failJob(ctx, db, p.JobID, "credentials not configured")
			return nil
		}

		// ── 3. Fetch library from adapter ─────────────────────────────────
		var entries []ExternalLibraryEntry
		var fetchErr error
		switch p.Storefront {
		case "steam":
			var creds struct {
				WebAPIKey string `json:"web_api_key"`
				SteamID   string `json:"steam_id"`
			}
			if err := json.Unmarshal([]byte(*cfg.StorefrontCredentials), &creds); err != nil {
				failJob(ctx, db, p.JobID, "invalid steam credentials")
				return nil
			}
			entries, fetchErr = steam.GetOwnedGames(ctx, creds.WebAPIKey, creds.SteamID)
		case "psn":
			var creds struct {
				NpssoToken string `json:"npsso_token"`
				IsVerified bool   `json:"is_verified"`
			}
			if err := json.Unmarshal([]byte(*cfg.StorefrontCredentials), &creds); err != nil {
				failJob(ctx, db, p.JobID, "invalid psn credentials")
				return nil
			}
			if !creds.IsVerified {
				failJob(ctx, db, p.JobID, "psn_token_expired")
				return nil
			}
			entries, fetchErr = psnLib.GetLibrary(ctx, creds.NpssoToken)
			if fetchErr != nil {
				// Mark token expired in credentials.
				expiredNow := time.Now().UTC()
				newCreds := map[string]any{
					"npsso_token":      creds.NpssoToken,
					"is_verified":      false,
					"token_expired_at": expiredNow,
				}
				if b, err := json.Marshal(newCreds); err == nil {
					s := string(b)
					_, _ = db.NewRaw(
						`UPDATE user_sync_configs SET storefront_credentials = ?, updated_at = now() WHERE user_id = ? AND storefront = 'psn'`,
						s, p.UserID,
					).Exec(context.Background())
				}
				failJob(ctx, db, p.JobID, "psn_token_expired")
				return nil
			}
		default:
			failJob(ctx, db, p.JobID, fmt.Sprintf("unknown storefront: %s", p.Storefront))
			return nil
		}
		if fetchErr != nil {
			failJob(ctx, db, p.JobID, fmt.Sprintf("fetch library: %v", fetchErr))
			return nil
		}

		// ── 4. Upsert ExternalGame rows ───────────────────────────────────
		fetchedIDs := make(map[string]struct{}, len(entries))
		for _, e := range entries {
			fetchedIDs[e.ExternalID] = struct{}{}
			row := &models.ExternalGame{
				ID:              uuid.NewString(),
				UserID:          p.UserID,
				Storefront:      p.Storefront,
				ExternalID:      e.ExternalID,
				Title:           e.Title,
				IsAvailable:     true,
				IsSubscription:  e.IsSubscription,
				PlaytimeHours:   e.PlaytimeHours,
				OwnershipStatus: &e.OwnershipStatus,
				CreatedAt:       time.Now().UTC(),
				UpdatedAt:       time.Now().UTC(),
			}
			_, _ = db.NewInsert().Model(row).
				On("CONFLICT (user_id, storefront, external_id) DO UPDATE SET title = EXCLUDED.title, playtime_hours = EXCLUDED.playtime_hours, is_subscription = EXCLUDED.is_subscription, ownership_status = EXCLUDED.ownership_status, is_available = true, updated_at = now()").
				Exec(ctx)
		}

		// ── 5. Mark removed games ─────────────────────────────────────────
		var available []models.ExternalGame
		_ = db.NewSelect().Model(&available).Where("user_id = ? AND storefront = ? AND is_available = true", p.UserID, p.Storefront).Scan(ctx)
		for _, eg := range available {
			if _, found := fetchedIDs[eg.ExternalID]; !found {
				_, _ = db.NewRaw(`UPDATE external_games SET is_available = false, updated_at = now() WHERE id = ?`, eg.ID).Exec(ctx)
				if eg.IsSubscription {
					_, _ = db.NewRaw(
						`UPDATE user_game_platforms SET ownership_status = 'no_longer_owned' WHERE external_game_id = ?`,
						eg.ID,
					).Exec(context.Background())
				}
			}
		}

		// ── 6. Dispatch ProcessSyncItemTask for each available, non-skipped game ──
		var toProcess []models.ExternalGame
		_ = db.NewSelect().Model(&toProcess).Where("user_id = ? AND storefront = ? AND is_available = true AND is_skipped = false", p.UserID, p.Storefront).Scan(ctx)

		// Build a map from external_id to raw_platform from this fetch.
		rawPlatformByExtID := make(map[string]string, len(entries))
		for _, e := range entries {
			rawPlatformByExtID[e.ExternalID] = e.RawPlatform
		}

		// Create a job_items batch for ProcessSyncItemTask.
		for _, eg := range toProcess {
			rawPlatform := rawPlatformByExtID[eg.ExternalID]
			meta := map[string]string{"external_game_id": eg.ID, "raw_platform": rawPlatform}
			metaJSON, _ := json.Marshal(meta)

			itemID := uuid.NewString()
			_, _ = db.NewRaw(
				`INSERT INTO job_items (id, job_id, user_id, item_key, source_title, source_metadata, status, result, igdb_candidates, created_at)
				 VALUES (?, ?, ?, ?, ?, ?, 'pending', '{}', '[]', now())`,
				itemID, p.JobID, p.UserID, eg.ExternalID, eg.Title, string(metaJSON),
			).Exec(ctx)

			_, _ = db.NewRaw(
				`INSERT INTO pending_tasks (id, task_type, payload, priority, status, attempts, created_at)
				 VALUES (?, 'process_sync_item', ?, 5, 'pending', 0, now())`,
				uuid.NewString(),
				fmt.Sprintf(`{"job_item_id":"%s"}`, itemID),
			).Exec(ctx)
		}

		// ── 7. Update last_synced_at ──────────────────────────────────────
		syncedNow := time.Now().UTC()
		_, _ = db.NewRaw(
			`UPDATE user_sync_configs SET last_synced_at = ?, updated_at = now() WHERE user_id = ? AND storefront = ?`,
			syncedNow, p.UserID, p.Storefront,
		).Exec(context.Background())

		return nil
	}
}

import (
	// add to existing imports:
	"github.com/drzero42/nexorious-go/internal/services/igdb"
	"github.com/drzero42/nexorious-go/internal/services/matching"
	"github.com/drzero42/nexorious-go/internal/services/platformresolution"

// processSyncItemPayload is the PendingTask.Payload shape for "process_sync_item".
type processSyncItemPayload struct {
	JobItemID string `json:"job_item_id"`
}

// NewProcessSyncItemHandler returns a TaskHandler for processing a single external game:
// IGDB matching → UserGamePlatform creation → job completion check.
func NewProcessSyncItemHandler(db *bun.DB, igdbClient *igdb.Client) func(ctx context.Context, task *models.PendingTask) error {
	return func(ctx context.Context, task *models.PendingTask) error {
		var p processSyncItemPayload
		if err := json.Unmarshal(task.Payload, &p); err != nil {
			slog.Error("process_sync_item: unmarshal payload", "err", err)
			return nil
		}

		var item models.JobItem
		if err := db.NewSelect().Model(&item).Where("id = ?", p.JobItemID).Scan(ctx); err != nil {
			slog.Error("process_sync_item: load job_item", "id", p.JobItemID, "err", err)
			return nil
		}

		// ── Phase 0: parse source_metadata ───────────────────────────────
		var meta struct {
			ExternalGameID string `json:"external_game_id"`
			RawPlatform    string `json:"raw_platform"`
		}
		if err := json.Unmarshal(item.SourceMetadata, &meta); err != nil {
			syncMarkItemFailed(ctx, db, &item, fmt.Sprintf("parse source_metadata: %v", err))
			syncCheckJobCompletion(ctx, db, item.JobID)
			return nil
		}

		var eg models.ExternalGame
		if err := db.NewSelect().Model(&eg).Where("id = ?", meta.ExternalGameID).Scan(ctx); err != nil {
			syncMarkItemFailed(ctx, db, &item, "external game not found")
			syncCheckJobCompletion(ctx, db, item.JobID)
			return nil
		}

		// ── Phase 1: skip if marked ───────────────────────────────────────
		if eg.IsSkipped {
			syncMarkItemSkipped(ctx, db, &item)
			syncCheckJobCompletion(ctx, db, item.JobID)
			return nil
		}

		// ── Phase 4: IGDB resolution ──────────────────────────────────────
		if eg.ResolvedIGDBID == nil && igdbClient != nil && igdbClient.Configured() {
			candidates, err := igdbClient.SearchGames(ctx, eg.Title)
			if err != nil {
				slog.Warn("process_sync_item: igdb search failed", "title", eg.Title, "err", err)
			} else {
				bestScore, bestID := 0.0, int32(0)
				for _, candidate := range candidates {
					score := matching.FuzzyScore(eg.Title, candidate.Name)
					if score > bestScore {
						bestScore = score
						bestID = int32(candidate.ID)
					}
				}
				if bestScore >= 0.85 {
					eg.ResolvedIGDBID = &bestID
					_, _ = db.NewRaw(`UPDATE external_games SET resolved_igdb_id = ?, updated_at = now() WHERE id = ?`, bestID, eg.ID).Exec(ctx)
					// Ensure the game row exists.
					_, _ = db.NewRaw(`INSERT INTO games (id, title, last_updated, created_at) VALUES (?, ?, now(), now()) ON CONFLICT (id) DO NOTHING`, bestID, eg.Title).Exec(ctx)
				} else {
					// Store candidates as JSON on the job item for review.
					candidatesJSON, _ := json.Marshal(candidates)
					item.IGDBCandidates = candidatesJSON
					syncMarkItemPendingReview(ctx, db, &item)
					syncCheckJobCompletion(ctx, db, item.JobID)
					return nil
				}
			}
		}

		if eg.ResolvedIGDBID == nil {
			syncMarkItemPendingReview(ctx, db, &item)
			syncCheckJobCompletion(ctx, db, item.JobID)
			return nil
		}

		// ── Phase 5: Collection sync ──────────────────────────────────────
		platformSlug, platformOK := platformresolution.RawPlatformToSlug(meta.RawPlatform)
		storefrontSlug, storefrontOK := platformresolution.StorefrontToCollectionSlug(eg.Storefront)
		if !platformOK || !storefrontOK {
			syncMarkItemFailed(ctx, db, &item, fmt.Sprintf("unresolved platform=%s storefront=%s", meta.RawPlatform, eg.Storefront))
			syncCheckJobCompletion(ctx, db, item.JobID)
			return nil
		}

		// Find or create UserGame.
		var ugID string
		err := db.NewRaw(`SELECT id FROM user_games WHERE user_id = ? AND game_id = ?`, item.UserID, *eg.ResolvedIGDBID).Scan(ctx, &ugID)
		if err != nil {
			ugID = uuid.NewString()
			now := time.Now().UTC()
			_, err = db.NewRaw(
				`INSERT INTO user_games (id, user_id, game_id, created_at, updated_at) VALUES (?, ?, ?, ?, ?) ON CONFLICT DO NOTHING RETURNING id`,
				ugID, item.UserID, *eg.ResolvedIGDBID, now, now,
			).Exec(ctx)
			if err != nil {
				// Fetch again in case it was created by a race.
				_ = db.NewRaw(`SELECT id FROM user_games WHERE user_id = ? AND game_id = ?`, item.UserID, *eg.ResolvedIGDBID).Scan(ctx, &ugID)
			}
		}

		// Find or create UserGamePlatform.
		var ugpID string
		err = db.NewRaw(
			`SELECT id FROM user_game_platforms WHERE external_game_id = ?`, eg.ID,
		).Scan(ctx, &ugpID)
		if err != nil {
			// Try by (user_game_id, platform, storefront).
			err = db.NewRaw(
				`SELECT id FROM user_game_platforms WHERE user_game_id = ? AND platform = ? AND storefront = ?`,
				ugID, platformSlug, storefrontSlug,
			).Scan(ctx, &ugpID)
		}

		ownershipStatus := "owned"
		if eg.OwnershipStatus != nil {
			ownershipStatus = *eg.OwnershipStatus
		}
		hoursPlayed := float64(eg.PlaytimeHours)

		if err != nil {
			// Create new.
			ugpID = uuid.NewString()
			now := time.Now().UTC()
			_, _ = db.NewRaw(
				`INSERT INTO user_game_platforms (id, user_game_id, platform, storefront, external_game_id, is_available, hours_played, ownership_status, sync_from_source, created_at, updated_at)
				 VALUES (?, ?, ?, ?, ?, true, ?, ?, true, ?, ?)`,
				ugpID, ugID, platformSlug, storefrontSlug, eg.ID, hoursPlayed, ownershipStatus, now, now,
			).Exec(ctx)
		} else {
			// Update — never downgrade ownership rank.
			ownershipRank := map[string]int{"owned": 4, "borrowed": 3, "rented": 3, "subscription": 2, "no_longer_owned": 1}
			var currentOwnership string
			_ = db.NewRaw(`SELECT COALESCE(ownership_status, '') FROM user_game_platforms WHERE id = ?`, ugpID).Scan(ctx, &currentOwnership)
			if ownershipRank[ownershipStatus] > ownershipRank[currentOwnership] {
				_, _ = db.NewRaw(`UPDATE user_game_platforms SET ownership_status = ?, updated_at = now() WHERE id = ?`, ownershipStatus, ugpID).Exec(ctx)
			}
			_, _ = db.NewRaw(`UPDATE user_game_platforms SET hours_played = ?, is_available = true, updated_at = now() WHERE id = ?`, hoursPlayed, ugpID).Exec(ctx)
		}

		syncMarkItemCompleted(ctx, db, &item)
		syncCheckJobCompletion(ctx, db, item.JobID)
		return nil
	}
}

// CheckPendingSyncs queries all non-manual sync configs and dispatches sync jobs for those overdue.
func CheckPendingSyncs(ctx context.Context, db *bun.DB, pool *worker.Pool) {
	var configs []models.UserSyncConfig
	if err := db.NewSelect().Model(&configs).Where("frequency != 'manual'").Scan(ctx); err != nil {
		slog.Error("CheckPendingSyncs: query configs", "err", err)
		return
	}

	now := time.Now().UTC()
	intervals := map[string]float64{
		"hourly": 3600,
		"daily":  86400,
		"weekly": 604800,
	}

	for _, cfg := range configs {
		if cfg.Storefront == "epic" {
			continue // no epic sync implementation
		}

		needsSync := false
		if cfg.LastSyncedAt == nil {
			needsSync = true
		} else if threshold, ok := intervals[cfg.Frequency]; ok {
			elapsed := now.Sub(*cfg.LastSyncedAt).Seconds()
			needsSync = elapsed >= threshold
		}

		if !needsSync {
			continue
		}

		// Check for existing active job.
		var existingID string
		err := db.NewRaw(
			`SELECT id FROM jobs WHERE user_id = ? AND job_type = 'sync' AND source = ? AND status IN ('pending', 'processing') LIMIT 1`,
			cfg.UserID, cfg.Storefront,
		).Scan(ctx, &existingID)
		if err == nil {
			continue // already running
		}

		// Create low-priority job.
		jobID := uuid.NewString()
		_, err = db.NewRaw(
			`INSERT INTO jobs (id, user_id, job_type, source, status, priority, created_at) VALUES (?, ?, 'sync', ?, 'pending', 'low', now())`,
			jobID, cfg.UserID, cfg.Storefront,
		).Exec(ctx)
		if err != nil {
			slog.Error("CheckPendingSyncs: insert job", "err", err)
			continue
		}

		_ = pool.Submit(ctx, "dispatch_sync", map[string]string{
			"job_id": jobID, "user_id": cfg.UserID, "storefront": cfg.Storefront,
		}, 1) // low priority
	}
}
